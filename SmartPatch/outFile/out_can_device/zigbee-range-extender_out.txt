def parse(String description) {
	def map = zigbee.getEvent(description)
	def result
	if(!map) {
		result = parseAttrMessage(description)
	} else {
		log.warn "Unexpected event: ${map}"
	}
	log.debug "Description ${description} parsed to ${result}"
	return result
}
